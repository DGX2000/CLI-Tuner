set(CPACK_PACKAGE_VENDOR "DGX2000")
set(CPACK_PACKAGE_NAME ${PROJECT_NAME})
set(CPACK_PACKAGE_CONTACT "DGX2000 <arnevoss@pm.me>")
set(HOMEPAGE "https://github.com/DGX2000/CLI-Tuner")
set(CPACK_INSTALL_PREFIX "/usr/bin")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "An instrument tuner for the command-line.")
set(CPACK_PACKAGE_DESCRIPTION_FILE "${PROJECT_SOURCE_DIR}/README.md")
set(CPACK_RESOURCE_FILE_LICENSE "${PROJECT_SOURCE_DIR}/LICENSE")
set(CPACK_PACKAGE_VERSION_MAJOR ${PROJECT_VERSION_MAJOR})
